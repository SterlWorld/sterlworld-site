const canvas = document.getElementById('matrix'); const ctx = canvas.getContext('2d'); canvas.width = window.innerWidth; canvas.height = window.innerHeight; const regularSymbols = 'アカサタナハマヤラワ0123456789∞Ωπ✧⊙∆'.split(''); const specialSymbols = ['𓂀','𓆣','𓋹','𓊹','𓇳','𓎛']; // Your special hieroglyphs const fontSize = 16; const columns = canvas.width / fontSize; const drops = Array(Math.floor(columns)).fill(1); function draw() { ctx.fillStyle = 'rgba(0, 0, 0, 0.1)'; ctx.fillRect(0, 0, canvas.width, canvas.height); ctx.font = `${fontSize}px monospace`; for (let i = 0; i < drops.length; i++) { let text; let isSpecial = false; // Decide randomly if this character is special (very rare, e.g. 1.5% chance) if (Math.random() < 0.015) { text = specialSymbols[Math.floor(Math.random() * specialSymbols.length)]; isSpecial = true; } else { text = regularSymbols[Math.floor(Math.random() * regularSymbols.length)]; } const x = i * fontSize; const y = drops[i] * fontSize; if (isSpecial) { // Gold glowing style for special symbols ctx.fillStyle = 'gold'; ctx.shadowColor = 'gold'; ctx.shadowBlur = 15; } else { // Regular cosmic blue-green style const green = Math.floor(150 + Math.random() * 100); const blue = Math.floor(180 + Math.random() * 75); ctx.fillStyle = `rgba(0, ${green}, ${blue}, 0.8)`; ctx.shadowBlur = 0; } ctx.fillText(text, x, y); if (y > canvas.height && Math.random() > 0.975) { drops[i] = 0; } drops[i]++; } } setInterval(draw, 50); // Keep canvas full screen on window resize window.addEventListener('resize', () => { canvas.width = window.innerWidth; canvas.height = window.innerHeight; });
